## Get started in three steps

Add Cossistant to your application and start providing AI-powered customer support instantly.

<Steps>
  <Step title="Install the package">
    Choose the package for your framework:

    <CodeGroup>
    ```bash Next.js
    npm install @cossistant/next
    # or
    pnpm add @cossistant/next
    # or
    bun add @cossistant/next
    ```

    ```bash React
    npm install @cossistant/react
    # or
    pnpm add @cossistant/react
    # or
    bun add @cossistant/react
    ```
    </CodeGroup>

  </Step>

  <Step title="Set your API key">
    Add your Cossistant public API key to your environment variables:

    ```bash .env.local
    NEXT_PUBLIC_COSSISTANT_API_KEY=pk_test_xxxx
    ```

    <Note>
      Get your API key from the [Cossistant dashboard](https://cossistant.com/dashboard). The SDK auto-detects environment variables: `NEXT_PUBLIC_COSSISTANT_API_KEY` (Next.js), `VITE_COSSISTANT_API_KEY` (Vite), or `COSSISTANT_API_KEY` (other).
    </Note>
  </Step>

  <Step title="Add the widget to your app">
    Wrap your app with `SupportProvider` and add the `Support` component:

    <CodeGroup>
    ```tsx Next.js (app/layout.tsx)
    import { SupportProvider } from "@cossistant/next";
    import type { Metadata } from "next";

    export const metadata: Metadata = {
      title: "Your App",
      description: "Your app description",
    };

    export default function RootLayout({
      children,
    }: {
      children: React.ReactNode;
    }) {
      return (
        <html lang="en">
          <body>
            <SupportProvider>{children}</SupportProvider>
          </body>
        </html>
      );
    }
    ```

    ```tsx Next.js (app/page.tsx)
    import { Support } from "@cossistant/next";

    export default function HomePage() {
      return (
        <main>
          <h1>Welcome to your app</h1>
          {/* Your page content */}
          
          <Support />
        </main>
      );
    }
    ```

    ```tsx React
    import { SupportProvider, Support } from "@cossistant/react";
    import "@cossistant/react/styles.css";

    export function App() {
      return (
        <SupportProvider>
          <div>
            <h1>Welcome to your app</h1>
            {/* Your app content */}
          </div>
          
          <Support />
        </SupportProvider>
      );
    }
    ```
    </CodeGroup>

    <Note>
      The `<Support />` component renders as a floating chat widget in the bottom-right corner of your app.
    </Note>
  </Step>
</Steps>

## Import styles (React only)

If you're using `@cossistant/react`, you need to import CSS. Choose the option that matches your setup:

<Tabs>
  <Tab title="Plain CSS">
    Import the pre-compiled CSS bundle (no Tailwind dependency):

    ```tsx
    import "@cossistant/react/styles.css";
    ```

    This works in any React app without requiring Tailwind CSS.
  </Tab>

  <Tab title="Tailwind v4">
    If you're using Tailwind CSS v4, import the source file for full theme customization:

    ```tsx
    import "@cossistant/react/support.css";
    ```

    <Warning>
      Tailwind v3 is not supported. Use the plain CSS import if you're on Tailwind v3.
    </Warning>
  </Tab>
</Tabs>

<Note>
  **Next.js users**: `@cossistant/next` includes CSS automatically—no manual import needed.
</Note>

## Verify it works

Start your development server and you should see the Cossistant chat widget in the bottom-right corner:

<CodeGroup>
```bash Next.js
npm run dev
```

```bash Vite
npm run dev
```

```bash Create React App
npm start
```
</CodeGroup>

Click the widget to open the chat interface. You're now ready to receive customer messages!

## Common customizations

### Identify logged-in users

Attach user information to support conversations so your team knows who they're talking to:

<CodeGroup>
```tsx Next.js (app/dashboard/layout.tsx)
import { IdentifySupportVisitor } from "@cossistant/next";

export default async function DashboardLayout({
  children,
}: {
  children: React.ReactNode;
}) {
  const session = await getSession(); // Your auth solution

  return (
    <div>
      {session?.user && (
        <IdentifySupportVisitor
          email={session.user.email}
          externalId={session.user.id}
          name={session.user.name}
          image={session.user.image}
          metadata={{
            plan: session.user.plan,
            signupDate: session.user.createdAt,
          }}
        />
      )}
      {children}
    </div>
  );
}
```

```tsx React
import { IdentifySupportVisitor } from "@cossistant/react";

export function Dashboard({ user }: { user: User }) {
  return (
    <>
      <IdentifySupportVisitor
        email={user.email}
        externalId={user.id}
        name={user.name}
        image={user.image}
        metadata={{
          plan: user.plan,
          signupDate: user.createdAt,
        }}
      />
      {/* Your dashboard content */}
    </>
  );
}
```
</CodeGroup>

<Note>
  Make sure `IdentifySupportVisitor` is rendered inside `SupportProvider` and the `<Support />` component is mounted somewhere in that tree.
</Note>

### Add page-specific quick options

Show contextual quick reply options based on the current page:

<CodeGroup>
```tsx Next.js (app/pricing/page.tsx)
import { SupportConfig } from "@cossistant/next";

export default function PricingPage() {
  return (
    <main>
      <SupportConfig
        quickOptions={[
          "How does billing work?",
          "What's included in Pro?",
          "Can I cancel anytime?",
        ]}
      />
      <h1>Pricing</h1>
      {/* Your pricing content */}
    </main>
  );
}
```

```tsx React
import { SupportConfig } from "@cossistant/react";

export function PricingPage() {
  return (
    <>
      <SupportConfig
        quickOptions={[
          "How does billing work?",
          "What's included in Pro?",
          "Can I cancel anytime?",
        ]}
      />
      <div>
        <h1>Pricing</h1>
        {/* Your pricing content */}
      </div>
    </>
  );
}
```
</CodeGroup>

### Display custom welcome messages

Show different welcome messages on different pages:

```tsx
import { SupportConfig, SenderType } from "@cossistant/react";

export function Dashboard() {
  return (
    <>
      <SupportConfig
        defaultMessages={[
          {
            content: "Welcome to your dashboard. If you need any help, I'm here!",
            senderType: SenderType.TeamMember,
          },
        ]}
      />
      {/* Your dashboard content */}
    </>
  );
}
```

## Troubleshooting

<AccordionGroup>
  <Accordion title="Widget doesn't appear">
    - Verify your API key is set correctly in `.env.local`
    - Check that `SupportProvider` wraps your entire app
    - Ensure the `<Support />` component is rendered
    - Check the browser console for errors
  </Accordion>

  <Accordion title="Styles not loading (React only)">
    - Make sure you imported the CSS: `import "@cossistant/react/styles.css"`
    - For Tailwind v4, use: `import "@cossistant/react/support.css"`
    - Verify your bundler is configured to handle CSS imports
  </Accordion>

  <Accordion title="TypeScript errors">
    - Ensure you're using React 18+ and Next.js 13.5+
    - Check that `@types/react` matches your React version
    - Run `npm install` to ensure all peer dependencies are installed
  </Accordion>

  <Accordion title="Environment variable not detected">
    - Restart your dev server after adding environment variables
    - For Next.js, use `NEXT_PUBLIC_COSSISTANT_API_KEY`
    - For Vite, use `VITE_COSSISTANT_API_KEY`
    - You can also pass `publicKey` directly to `SupportProvider`
  </Accordion>
</AccordionGroup>
