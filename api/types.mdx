# TypeScript Types Reference

Complete TypeScript type definitions from `@cossistant/types`.

## Core Types

### CossistantConfig

Configuration object for initializing the SDK client.

```typescript
type CossistantConfig = {
  apiUrl: string;
  wsUrl: string;
  apiKey?: string;
  publicKey?: string;
  userId?: string;
  organizationId?: string;
};
```

### CossistantError

Standardized error object.

```typescript
type CossistantError = {
  code: string;
  message: string;
  details?: Record<string, unknown>;
};
```

### DefaultMessage

Pre-filled messages for new conversations.

```typescript
type DefaultMessage = {
  content: string;
  senderType: SenderType;
  senderId?: string;
};
```

## Timeline Item Types

### TimelineItem

Base timeline item structure (AI SDK v6 compatible).

```typescript
type TimelineItem = {
  id: string;
  conversationId: string;
  organizationId: string;
  visibility: TimelineItemVisibility;
  type: ConversationTimelineType;
  text: string | null;
  tool?: string | null;
  parts: TimelineItemParts;
  userId: string | null;
  aiAgentId: string | null;
  visitorId: string | null;
  createdAt: string;
  deletedAt?: string | null;
};
```

### TimelineItemParts

AI SDK v6 compatible parts array.

```typescript
type TimelineItemParts = Array<
  | TextPart
  | ReasoningPart
  | ToolPart
  | SourceUrlPart
  | SourceDocumentPart
  | StepStartPart
  | FilePart
  | ImagePart
  | TimelinePartEvent
  | TimelinePartMetadata
>;
```

### TextPart

Plain text content.

```typescript
type TextPart = {
  type: "text";
  text: string;
  state?: "streaming" | "done";
};
```

### ReasoningPart

```typescript
type ReasoningPart = {
  type: "reasoning";
```

## Type Guards

Helper functions for type narrowing.

```typescript
function isTextPart(part: TimelineItemParts[number]): part is TextPart {
  return part.type === "text";
}

function isToolPart(part: TimelineItemParts[number]): part is ToolPart {
  return part.type.startsWith("tool-");
}

function isFilePart(part: TimelineItemParts[number]): part is FilePart {
  return part.type === "file";
}

function isImagePart(part: TimelineItemParts[number]): part is ImagePart {
  return part.type === "image";
}
```
