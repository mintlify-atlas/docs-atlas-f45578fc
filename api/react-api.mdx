# React SDK API Reference

Complete reference for all exports from `@cossistant/react`.

## Provider

### SupportProvider

Root provider component that initializes the Cossistant SDK and manages global state.

```tsx
import { SupportProvider } from "@cossistant/react";

function App() {
  return (
    <SupportProvider publicKey="pk_...">
      {/* Your app */}
    </SupportProvider>
  );
}
```

<ParamField path="publicKey" type="string" required>
  Your Cossistant public API key
</ParamField>

<ParamField path="apiUrl" type="string" default="https://api.cossistant.com/v1">
  Override the API URL (useful for self-hosted instances)
</ParamField>

<ParamField path="wsUrl" type="string" default="wss://api.cossistant.com/ws">
  Override the WebSocket URL for realtime features
</ParamField>

<ParamField path="defaultMessages" type="DefaultMessage[]">
  Pre-fill the conversation with welcome messages before visitor interaction
</ParamField>

<ParamField path="quickOptions" type="string[]">
  Quick reply buttons displayed to visitors
</ParamField>

<ParamField path="autoConnect" type="boolean" default={true}>
  Whether to automatically establish WebSocket connection
</ParamField>

<ParamField path="size" type="'normal' | 'larger'" default="normal">
  Widget display size
</ParamField>

<ParamField path="defaultOpen" type="boolean" default={false}>
  Whether the widget should open automatically on mount
</ParamField>

<ParamField path="onWsConnect" type="() => void">
  Callback fired when WebSocket connection is established
</ParamField>

<ParamField path="onWsDisconnect" type="() => void">
  Callback fired when WebSocket connection is lost
</ParamField>

<ParamField path="onWsError" type="(error: Error) => void">
  Callback fired when WebSocket errors occur
</ParamField>

## Components

### Support

Complete support widget with chat, routing, and real-time features.

```tsx
import { Support } from "@cossistant/react";

<Support />
```

#### Props

<ParamField path="side" type="'top' | 'bottom' | 'left' | 'right'" default="top">
  Position of the widget relative to the trigger
</ParamField>

<ParamField path="align" type="'start' | 'center' | 'end'" default="end">
  Alignment along the side axis
</ParamField>

<ParamField path="sideOffset" type="number" default={16}>
  Distance in pixels between trigger and content
</ParamField>

<ParamField path="avoidCollisions" type="boolean" default={true}>
  Enable automatic repositioning to stay within viewport
</ParamField>

<ParamField path="collisionPadding" type="number | { top?: number; right?: number; bottom?: number; left?: number }" default={8}>
  Padding from viewport edges when avoiding collisions
</ParamField>

<ParamField path="theme" type="'light' | 'dark'">
  Force a specific theme (omit for automatic detection)
</ParamField>

<ParamField path="open" type="boolean">
  Controlled open state - use with `onOpenChange` for external state management
</ParamField>

<ParamField path="onOpenChange" type="(open: boolean) => void">
  Callback fired when open state should change (for controlled mode)
</ParamField>

<ParamField path="defaultOpen" type="boolean" default={false}>
  Whether widget opens automatically in uncontrolled mode
</ParamField>

<ParamField path="quickOptions" type="string[]">
  Quick reply buttons displayed to users
</ParamField>

<ParamField path="defaultMessages" type="DefaultMessage[]">
  Welcome messages shown before conversation starts
</ParamField>

<ParamField path="locale" type="string">
  Locale string for widget translations (e.g., 'en', 'fr', 'es')
</ParamField>

<ParamField path="content" type="SupportTextContentOverrides">
  Custom text content overrides for internationalization
</ParamField>

<ParamField path="customPages" type="CustomPage[]">
  Custom pages to add to the router
</ParamField>

<ParamField path="className" type="string">
  Additional CSS classes for root wrapper
</ParamField>

<ParamField path="classNames" type="{ trigger?: string; content?: string }">
  CSS class overrides for specific parts
</ParamField>

#### Event Callbacks

<ParamField path="onConversationStart" type="(event: ConversationStartEvent) => void">
  Called when a new conversation is started
</ParamField>

<ParamField path="onConversationEnd" type="(event: ConversationEndEvent) => void">
  Called when a conversation ends (resolved, closed, etc.)
</ParamField>

<ParamField path="onMessageSent" type="(event: MessageSentEvent) => void">
  Called when the visitor sends a message
</ParamField>

<ParamField path="onMessageReceived" type="(event: MessageReceivedEvent) => void">
  Called when a message is received from an agent
</ParamField>

<ParamField path="onError" type="(event: ErrorEvent) => void">
  Called when an error occurs within the widget
</ParamField>

#### Compound Components

<CodeGroup>
```tsx Support.Root
<Support.Root open={isOpen} onOpenChange={setIsOpen}>

  <Support.Content>
    <Support.Router />

```

```tsx Support.Trigger
<Support.Trigger className="custom-button">
  {({ isOpen, unreadCount }) => (
    <span>{isOpen ? 'Close' : `Help (${unreadCount})`}</span>

```

```tsx Support.Content

  <Support.Router />
</Support.Content>
```

```tsx Support.Page

  <Support.Page name="FAQ" component={FAQPage} />
</Support>
```

```tsx Support.Router
<Support.Router>
  <Support.Page name="custom" component={CustomPage} />
</Support.Router>
```

## Hooks

### useSupport

Access the global support context and website data.

```tsx
import { useSupport } from "@cossistant/react";

function MyComponent() {
  const {
    website,
    visitor,
    client,
    unreadCount,
    isOpen,
    open,
    close,
    toggle,
    availableHumanAgents,
    availableAIAgents
  } = useSupport();
}
```

#### Returns

<ResponseField name="website" type="PublicWebsiteResponse | null">
  Current website configuration and settings
</ResponseField>

<ResponseField name="visitor" type="PublicVisitor | undefined">
  Current visitor information with locale
</ResponseField>

<ResponseField name="client" type="CossistantClient | null">
  SDK client instance for direct API calls
</ResponseField>

<ResponseField name="unreadCount" type="number">
  Number of unread messages across all conversations
</ResponseField>

<ResponseField name="isOpen" type="boolean">
  Current open state of the support widget
</ResponseField>

<ResponseField name="open" type="() => void">
  Function to open the support widget
</ResponseField>

<ResponseField name="close" type="() => void">
  Function to close the support widget
</ResponseField>

<ResponseField name="toggle" type="() => void">
  Function to toggle the support widget open/closed
</ResponseField>

<ResponseField name="availableHumanAgents" type="Agent[]">
  List of available human agents
</ResponseField>

<ResponseField name="availableAIAgents" type="AIAgent[]">
  List of available AI agents
</ResponseField>

<ResponseField name="isLoading" type="boolean">
  Whether initial data is loading
</ResponseField>

<ResponseField name="error" type="Error | null">
  Any error that occurred during initialization
</ResponseField>

<ResponseField name="configurationError" type="ConfigurationError | null">
  Configuration-specific errors (e.g., missing API key)

### useSendMessage

Send messages with optimistic updates and file upload support.

```tsx
import { useSendMessage } from "@cossistant/react";

const { mutate, mutateAsync, isPending, isUploading, error, reset } = useSendMessage();

// Fire and forget
mutate({
  conversationId: 'conv_123',
  message: 'Hello!',
  files: [file1, file2]
});

// Await result
const result = await mutateAsync({
  message: 'Hello!',
  onSuccess: (conversationId, messageId) => {
    console.log('Message sent:', messageId);
  }
});
```

#### Returns

<ResponseField name="mutate" type="(options: SendMessageOptions) => void">
  Fire-and-forget mutation (errors are swallowed)
</ResponseField>

<ResponseField name="mutateAsync" type="(options: SendMessageOptions) => Promise<SendMessageResult | null>">
  Async mutation that returns a promise
  
  <Expandable title="SendMessageOptions">
    <ParamField path="conversationId" type="string">
      Existing conversation ID (omit to create new conversation)
    </ParamField>
    <ParamField path="message" type="string" required>
      Message text content
    </ParamField>
    <ParamField path="files" type="File[]">
      Attachments to upload
    </ParamField>
    <ParamField path="messageId" type="string">
      Optional custom message ID
    </ParamField>
    <ParamField path="defaultTimelineItems" type="TimelineItem[]">
      Pre-fill timeline for new conversations
    </ParamField>
    <ParamField path="visitorId" type="string">
      Override visitor ID
    </ParamField>
    <ParamField path="onSuccess" type="(conversationId: string, messageId: string) => void">
      Success callback
    </ParamField>
    <ParamField path="onError" type="(error: Error) => void">
      Error callback
    </ParamField>
    <ParamField path="onConversationInitiated" type="(conversationId: string) => void">
      Called immediately when new conversation is initiated (before API call)
    </ParamField>
  </Expandable>
</ResponseField>

<ResponseField name="isPending" type="boolean">
  Whether the mutation is in progress
</ResponseField>

<ResponseField name="isUploading" type="boolean">
  Whether files are being uploaded
</ResponseField>

<ResponseField name="error" type="Error | null">
  Any error that occurred
</ResponseField>

<ResponseField name="reset" type="() => void">
  Reset error and loading states
</ResponseField>

## Events

Type-safe event system for support widget.

### useSupportEvents

```tsx
import { useSupportEvents } from "@cossistant/react";

const events = useSupportEvents();
events.on('conversationStart', (event) => {
  console.log('Conversation started:', event);
});
```

### Event Types

<CodeGroup>
```typescript ConversationStartEvent
type ConversationStartEvent = {
  conversationId: string;
  timestamp: string;
};
```

```typescript ConversationEndEvent
type ConversationEndEvent = {
  conversationId: string;
  status: 'resolved' | 'spam';
  timestamp: string;
};
```

```typescript MessageSentEvent
type MessageSentEvent = {
  conversationId: string;
  messageId: string;
  content: string;
  timestamp: string;
};
```

```typescript MessageReceivedEvent
type MessageReceivedEvent = {
  conversationId: string;
  messageId: string;
  content: string;
  senderType: 'ai' | 'team_member';
  timestamp: string;
};
```

```typescript ErrorEvent
type ErrorEvent = {
  error: Error;
  context?: string;
  timestamp: string;
};
```
</CodeGroup>
